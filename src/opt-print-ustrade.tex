\def\OPTmodes{twoside,final,openright,10pt}

% 经过一番尝试，以下是计算方法：
% 1. 使用计算器 (calculator) 获取你的规格 (specifications)
%    https://www.blurb.com/make/pdf_to_book/booksize_calculator
% 2. 获取规格 (specifications)。对于一本 6x9 硬壳封面 (hardcover) (ImageWrap) 书籍，我得到：
%
%    页面规格 (Page Specifications)	                    英寸 (Inches)
%    最终导出的 PDF (Final, exported PDF) 应为 (宽 x 高)	6.125 x 9.25
%    页面尺寸 (Page size) / 裁切线 (trim line) (宽 x 高)	            6.0 x 9.0
%    出血 (Bleed) (上、下及外边缘)	    0.125
%    安全边界 (Safe boundary) (上、下、外边缘)	0.25
%    安全边界 (Safe boundary) (装订边缘 (Binding Edge))	            0.5
%
% 3. 裁切线 (Trim line) 位于 0.625 英寸处，因此要获得安全的文本边界 (safe text boundary)（洋红色线条 (magenta line)）：
%    上边距 (Top margin):      0.625 英寸 (从裁切线 (trim line) 开始，页眉空间 (header space) 填充至安全线 (safe line))
%    下边距 (Bottom margin):   0.625 英寸 + 0.25 英寸 = 0.875 英寸
%    内边距 (Inner margin):    0.625 英寸 (从裁切线 (trim line) 开始)
%    装订偏移 (Binding offset):  0.5 英寸
%    外边距 (Outer margin):    0.625 英寸 + 0.25 英寸 = 0.875 英寸

% Blurb.com 规格 (Specifications)
\def\OPTpaperesize{6.125in,9.25in}     % 页面尺寸 (Page size)
\def\OPTtopmargin{.625in}              % 页面顶部的边距 (Margin at the top of the page)
\def\OPTbottommargin{0.875in}          % 页面底部的边距 (Margin at the bottom of the page)
\def\OPTinnermargin{0.625in}           % 页面内侧的边距 (Margin on the inner side of the page)
\def\OPTbindingoffset{0.4in}           % 内侧的额外偏移量 (Extra offset on the inner side)
\def\OPToutermargin{0.625in}           % 页面外侧的边距 (Margin on the outer side of the page)
